    # Imports n√©cessaires
import tensorflow as tf
from tensorflow.keras.datasets import imdb
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Embedding, LSTM, Dense, Dropout
from tensorflow.keras.preprocessing import sequence
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.manifold import TSNE
import pandas as pd

# Configuration
print(f"TensorFlow version: {tf.__version__}")
print("GPU disponible:", "Oui" if tf.config.list_physical_devices('GPU') else "Non")

# Param√®tres du mod√®le
MAX_FEATURES = 5000  # Nombre de mots dans le vocabulaire
MAX_LEN = 200       # Longueur maximale des s√©quences
EMBEDDING_SIZE = 128 # Taille des embeddings

print("\n‚úÖ Configuration termin√©e !")

    Cellule 3 (Code) - Chargement et exploration des donn√©es

    # Chargement du dataset IMDB (avis de films)
print("üì• Chargement du dataset IMDB...")
(X_train, y_train), (X_test, y_test) = imdb.load_data(num_words=MAX_FEATURES)

print(f"üìä Donn√©es d'entra√Ænement : {len(X_train)} avis")
print(f"üìä Donn√©es de test : {len(X_test)} avis")
print(f"üìä Vocabulaire : {MAX_FEATURES} mots les plus fr√©quents")

# R√©cup√©ration du dictionnaire de mots
word_index = imdb.get_word_index()
reverse_word_index = {v: k for k, v in word_index.items()}
reverse_word_index[0] = '<PAD>'
reverse_word_index[1] = '<START>'
reverse_word_index[2] = '<UNKNOWN>'

def decode_review(encoded_review):
    """Convertit une s√©quence d'indices en texte lisible"""
    return ' '.join([reverse_word_index.get(i, '<UNKNOWN>') for i in encoded_review])

# Affichage de quelques exemples
print("\nüìù Exemples d'avis (avant preprocessing) :")
for i in range(3):
    sentiment = "üòä POSITIF" if y_train[i] == 1 else "üòû N√âGATIF"
    print(f"\n{sentiment} - Longueur: {len(X_train[i])} mots")
    decoded = decode_review(X_train[i])
    # Afficher seulement les premiers mots pour la lisibilit√©
    print(f"Texte: {decoded[:200]}...")

# Distribution des longueurs
lengths = [len(x) for x in X_train]
plt.figure(figsize=(12, 4))

plt.subplot(1, 2, 1)
plt.hist(lengths, bins=50, alpha=0.7)
plt.title('Distribution des longueurs d\'avis')
plt.xlabel('Nombre de mots')
plt.ylabel('Fr√©quence')
plt.axvline(MAX_LEN, color='red', linestyle='--', label=f'Limite fix√©e: {MAX_LEN}')
plt.legend()

plt.subplot(1, 2, 2)
plt.hist(y_train, bins=2, alpha=0.7, color=['red', 'green'])
plt.title('Distribution des sentiments')
plt.xlabel('Sentiment (0=N√©gatif, 1=Positif)')
plt.ylabel('Nombre d\'avis')
plt.xticks([0, 1], ['N√©gatif', 'Positif'])

plt.tight_layout()
plt.show()

print(f"\nüìä Statistiques des longueurs :")
print(f"   - Longueur moyenne : {np.mean(lengths):.1f} mots")
print(f"   - Longueur m√©diane : {np.median(lengths):.1f} mots")
print(f"   - Avis > {MAX_LEN} mots : {np.sum(np.array(lengths) > MAX_LEN)}")

    Cellule 4 (Code) - Pr√©paration des donn√©es

    # Padding des s√©quences (toutes √† la m√™me longueur)
print("üîß Pr√©paration des donn√©es...")
print(f"   - Troncature/padding √† {MAX_LEN} mots")

X_train = sequence.pad_sequences(X_train, maxlen=MAX_LEN, padding='post')
X_test = sequence.pad_sequences(X_test, maxlen=MAX_LEN, padding='post')

print(f"   - Forme finale X_train : {X_train.shape}")
print(f"   - Forme finale X_test : {X_test.shape}")

# Visualisation de l'effet du padding
exemple_idx = 0

# Conversion en format appropri√© pour TensorFlow
X_train = X_train.astype('int32')
X_test = X_test.astype('int32')
y_train = y_train.astype('int32') 
y_test = y_test.astype('int32')

print("\n‚úÖ Donn√©es pr√©par√©es pour l'entra√Ænement !")

    Cellule 5 (Code) - Construction du mod√®le LSTM

    # Construction du mod√®le LSTM
print("üèóÔ∏è Construction du mod√®le LSTM...")

model = Sequential([
    # Couche d'embedding : convertit les indices en vecteurs denses
    Embedding(
        input_dim=MAX_FEATURES,    # Taille du vocabulaire
        output_dim=EMBEDDING_SIZE, # Dimension des embeddings
        input_length=MAX_LEN,      # Longueur des s√©quences
        name='embedding'
    ),

    # Couche LSTM : traite les s√©quences
    LSTM(
        units=64,           # Nombre d'unit√©s LSTM
        dropout=0.2,        # Dropout sur les entr√©es
        recurrent_dropout=0.2,  # Dropout sur les connexions r√©currentes
        name='lstm'
    ),

    # Couche de r√©gularisation
    Dropout(0.5, name='dropout'),

    # Couche de sortie : classification binaire
    Dense(1, activation='sigmoid', name='output')
])

# Compilation du mod√®le
model.compile(
    optimizer='adam',
    loss='binary_crossentropy',
    metrics=['accuracy']
)

# Affichage de l'architecture
print("\nüìã Architecture du mod√®le :")
model.summary()

print(f"\nüî¢ D√©tails des couches :")
print(f"   - Embedding : {MAX_FEATURES} mots ‚Üí {EMBEDDING_SIZE} dimensions")
print(f"   - LSTM : 64 unit√©s avec m√©moire s√©quentielle")
print(f"   - Dense : 1 neurone pour classification binaire (0-1)")
print(f"   - Total param√®tres : {model.count_params():,}")

    Cellule 6 (Code) - Entra√Ænement du mod√®le

    # Entra√Ænement du mod√®le

print("‚è±Ô∏è Les LSTM sont plus lents que les CNN, patience !")

# Entra√Ænement avec validation
history = model.fit(
    X_train, y_train,
    batch_size=128,
    epochs=3,  # Peu d'√©poques pour la d√©monstration
    validation_split=0.2,
    verbose=1
)

# √âvaluation finale
test_loss, test_accuracy = model.evaluate(X_test, y_test, verbose=0)
print(f"\nüìä R√©sultats finaux :")
print(f"   - Pr√©cision sur test : {test_accuracy*100:.2f}%")
print(f"   - Perte sur test : {test_loss:.4f}")

# Visualisation des courbes d'apprentissage
plt.figure(figsize=(15, 5))

plt.subplot(1, 2, 1)
plt.plot(history.history['accuracy'], 'b-', label='Entra√Ænement', linewidth=2)

plt.title('√âvolution de la pr√©cision')
plt.xlabel('√âpoque')
plt.ylabel('Pr√©cision')
plt.legend()
plt.grid(True, alpha=0.3)

plt.subplot(1, 2, 2)
plt.plot(history.history['loss'], 'b-', label='Entra√Ænement', linewidth=2)

    # Fonction pour tester des phrases personnalis√©es
def predict_sentiment(text, model, word_index, max_features=MAX_FEATURES, max_len=MAX_LEN):
    """
    Pr√©dit le sentiment d'un texte personnalis√©
    """
    # Pr√©traitement du texte
    words = text.lower().split()

    # Conversion en indices
    sequence = []
    for word in words:
        if word in word_index and word_index[word] < max_features:
            sequence.append(word_index[word])
        else:
            sequence.append(2)  # <UNKNOWN>

    # Padding
    if len(sequence) < max_len:
        sequence = sequence + [0] * (max_len - len(sequence))
    else:
        sequence = sequence[:max_len]

    # Pr√©diction
    sequence = np.array([sequence])
    prediction = model.predict(sequence, verbose=0)[0][0]

    sentiment = "üòä POSITIF" if prediction > 0.5 else "üòû N√âGATIF"
    confidence = prediction if prediction > 0.5 else 1 - prediction

    return sentiment, confidence, prediction

# Test avec des phrases d'exemple

    "The cinematography was beautiful but the story was confusing",

    "This film changed my life incredible experience"
]

print("\nüì± R√©sultats des tests :")

    print(f"\n{i}. \"{sentence}\"")
    print(f"   ‚Üí {sentiment} (confiance: {confidence:.1%}, score brut: {raw_score:.3f})")

    # Analyse des erreurs du mod√®le

y_pred = (y_pred_proba > 0.5).astype(int)

# Indices des erreurs
wrong_predictions = np.where(y_test != y_pred)[0]

print(f"üìä Statistiques d'erreurs :")
print(f"   - Erreurs totales : {len(wrong_predictions)}/{len(y_test)} ({len(wrong_predictions)/len(y_test):.1%})")

# Analyser les types d'erreurs
false_positives = np.where((y_test == 0) & (y_pred == 1))[0]  # Pr√©dit positif alors que c'est n√©gatif
false_negatives = np.where((y_test == 1) & (y_pred == 0))[0]  # Pr√©dit n√©gatif alors que c'est positif

print(f"   - Faux positifs : {len(false_positives)} (n√©gatifs class√©s comme positifs)")
print(f"   - Faux n√©gatifs : {len(false_negatives)} (positifs class√©s comme n√©gatifs)")

# Examiner quelques erreurs int√©ressantes
print(f"\nüîç Exemples d'erreurs int√©ressantes :")

# Faux positifs avec haute confiance
if len(false_positives) > 0:
    fp_confident = false_positives[np.argsort(y_pred_proba[false_positives])[-3:]]  # Top 3 plus confiants
    print(f"\n‚ùå Faux positifs (n√©gatifs pr√©dits comme positifs) :")
    for i, idx in enumerate(fp_confident):
        print(f"\n{i+1}. Confiance: {y_pred_proba[idx]:.1%}")
        decoded = decode_review(X_test[idx])
        clean_text = decoded.replace('<START>', '').replace('<PAD>', '').strip()
        words = clean_text.split()[:25]
        print(f"   Texte: {' '.join(words)}...")

# Faux n√©gatifs avec haute confiance
if len(false_negatives) > 0:
    fn_confident = false_negatives[np.argsort(y_pred_proba[false_negatives])[:3]]  # Top 3 moins confiants
    print(f"\n‚ùå Faux n√©gatifs (positifs pr√©dits comme n√©gatifs) :")
    for i, idx in enumerate(fn_confident):
        print(f"\n{i+1}. Confiance: {1-y_pred_proba[idx]:.1%}")
        decoded = decode_review(X_test[idx])
        clean_text = decoded.replace('<START>', '').replace('<PAD>', '').strip()
        words = clean_text.split()[:25]
        print(f"   Texte: {' '.join(words)}...")

# Test de cas difficiles

    "This movie is not bad",  # N√©gation
    "I expected it to be terrible but it was actually okay",  # Contraste
    "The worst movie ever... just kidding it was great",  # Sarcasme
    "So bad it's good",  # Paradoxe
    "Could have been better",  # Nuance
]

for case in difficult_cases:
    sentiment, confidence, raw_score = predict_sentiment(case, model, word_index)
    print(f"\nüìù \"{case}\"")
    print(f"   ‚Üí {sentiment} (confiance: {confidence:.1%})")
    print(f"   üí≠ Analyse: Cette phrase contient des nuances difficiles √† interpr√©ter")
